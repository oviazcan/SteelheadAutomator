' === MACRO: SombrearModoSoloPN (v13 — layout v12) ===
' Sombrea en gris las columnas que NO aplican según el modo.
' Se ejecuta automáticamente al cambiar A1 (Modo).
'
' V13 (2026-06-11): adaptado al layout v12 (60 cols visibles A..BH).
'   - Modo en A1 (antes H1 en v11). A2:H2 = título merged.
'   - Header de cotización: A3:M4 (la metadata llega a M por el merge J3:M4 de
'     Notas Internas; Q3:S4 es el helper "Combinación no existente", NO se sombrea).
'   - Nombre Cotización/Layout (quoteName) en F3 → permanece vivo en SOLO_PN.
'   - Cantidad: col I=9 (antes K=11). Rango I9:I508.
'   - Productos: ahora 1 combo en col V=22 (antes 3 grupos X..AI=24..35). Rango V9:V508.
' V12 (2026-05-27): H3 (nombre del layout) vivo en SOLO_PN. Strip A3:T4.
' V11: Modo H1, strip A3:T4, Cantidad K9:K508, Productos X9:AI508.
'
' Target: spreadsheet v12 layout, 60 cols A..BH, datos fila 9+

Sub SombrearModoSoloPN()
    Dim wsUp As Worksheet
    Set wsUp = ThisWorkbook.Sheets("Upload")

    ' V12: el modo está en A1
    Dim modo As String
    modo = UCase(Trim(wsUp.Range("A1").Value))

    Dim grisClaro As Long
    grisClaro = RGB(125, 125, 125)
    Dim verdeClaro As Long
    verdeClaro = RGB(232, 245, 233)

    ' Reset: pintar a verde (estado COTIZACIÓN+NP)
    ' Header de cotización v12: A3:M4
    wsUp.Range("A3:M4").Interior.Color = verdeClaro
    ' Datos: CANTIDAD (I=9) y PRODUCTOS (V=22)
    wsUp.Range("I9:I508").Interior.Color = verdeClaro
    wsUp.Range("V9:V508").Interior.Color = verdeClaro

    If InStr(modo, "SOLO") > 0 Then
        ' SOLO_PN: sombrear lo que NO aplica
        ' Header de cotización (Empresa, Notas, Asignado, Válida) sombreado;
        ' F3 (Nombre Cotización/Layout) siempre aplica → se restaura a verde.
        wsUp.Range("A3:M4").Interior.Color = grisClaro
        wsUp.Range("F3").Interior.Color = verdeClaro

        ' Cantidad (I=9): no aplica sin cotización
        wsUp.Range("I9:I508").Interior.Color = grisClaro
        ' Productos (V=22): no aplican sin cotización
        wsUp.Range("V9:V508").Interior.Color = grisClaro

        MsgBox "Modo S" & ChrW(243) & "lo N" & ChrW(250) & "meros de Parte:" & vbCrLf & _
               "- Header de cotizaci" & ChrW(243) & "n sombreado (F3 viva)" & vbCrLf & _
               "- Cantidad sombreada (no aplica)" & vbCrLf & _
               "- Productos sombreados (no aplican)", vbInformation, "Modo SOLO_PN"
    Else
        MsgBox "Modo Cotizaci" & ChrW(243) & "n + N" & ChrW(250) & "meros de Parte: todos los campos habilitados.", _
               vbInformation, "Modo COTIZACI" & ChrW(211) & "N+NP"
    End If
End Sub
