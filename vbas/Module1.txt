' === MACRO 1: ExportarCSV (v14) ===
' V14 (2026-05-28): predictive cols BD..BL (55..63) a NumberFormat "General"
'   en el libro temporal antes del SaveAs.
'   Bug origen: la plantilla muestra 4 decimales (legible para el usuario),
'   pero SaveAs FileFormat:=62 exporta el VALOR MOSTRADO, no el valor raw.
'   Con formato display "0.0000" el CSV salía con valores truncados y el
'   parser predictive de bulk-upload veía consumos demasiado bajos.
'   Forzar "General" en la copia temporal preserva precisión completa (hasta
'   15 dígitos significativos de Excel) sin tocar la UI visible al usuario.
' V13 (2026-05-26): lastRow considera col G (PN) y col E (Id SH) — toma el máximo
'   - Permite filas con solo Id SH (col E) al final del rango sin PN en col G
' V12 (2026-05-26): adaptado al spreadsheet layout v11 (71 cols, +Id SH, +Tipo de Geometría, Línea movida a col P)
'   - Modo: leer H1 (antes G1)
'   - quoteName: leer H3 (antes G3)
'   - Validación Cliente: col 6 (F) — antes col 5 (E)
'   - Validación PN:      col 7 (G) — antes col 6 (F)
'   - lastRow por col 7 (G = PN)
'   - Sort keys: F9 (Cliente) y G9 (PN) — antes E9 y F9
'   - SetRange sort: A9:BS — antes A9:BQ
'   - Cols extra: eliminar desde col 72 si UsedRange > 71 — antes >69 y col 70
' V11.1 (2026-05-20): ChrW(211) = Ó para sobrevivir paste-encoding del VBE
' V11 (2026-05-19): hardening para corrida masiva de 18k filas con bulk-upload
' V10: Exporta 69 columnas (A-BQ).
'
' Target: spreadsheet v11 layout, 71 cols A..BS, datos fila 9+

Sub ExportarCSV()
    Dim ws As Worksheet, csvName As String, savePath As String
    Set ws = ThisWorkbook.Sheets("Upload")

    ' V12: leer modo de H1 (antes G1) y quoteName de H3 (antes G3)
    Dim modo As String, quoteName As String
    modo = UCase(Trim(CStr(ws.Range("H1").Value)))
    quoteName = Trim(CStr(ws.Range("H3").Value))

    Dim modoNorm As String
    ' V11.1: ChrW(211) = Ó para sobrevivir paste-encoding del VBE
    modoNorm = Replace(Replace(modo, ChrW(211), "O"), " ", "_")
    If modoNorm <> "COTIZACION+NP" And modoNorm <> "SOLO_PN" Then
        MsgBox "Modo inv" & ChrW(225) & "lido en H1: '" & ws.Range("H1").Value & "'." & vbCrLf & vbCrLf & _
               "Debe ser 'COTIZACI" & ChrW(211) & "N+NP' o 'SOLO_PN'.", vbCritical, "Modo inv" & ChrW(225) & "lido"
        Exit Sub
    End If

    Dim esCotizacion As Boolean
    esCotizacion = (modoNorm = "COTIZACION+NP")

    If esCotizacion And quoteName = "" Then
        MsgBox "Modo COTIZACI" & ChrW(211) & "N+NP requiere 'Nombre Cotizaci" & ChrW(243) & "n' en H3.", _
               vbCritical, "QuoteName faltante"
        Exit Sub
    End If

    ' V12: validar que toda línea con PN (col G=7) tenga Cliente (col F=6)
    ' Cliente: necesario incluso en SOLO_PN porque los PNs están scoped por customer.
    ' Proceso: NO se valida aquí porque vacío tiene significado (PN existente: copiar el suyo).
    '          La extensión valida por línea y reporta errores sin matar el batch.
    Dim r As Long, lastR As Long, sinCliente As String, nFaltantes As Long
    Dim pnVal As String, clVal As String
    Dim lastG As Long, lastE As Long
    lastG = ws.Cells(ws.Rows.Count, 7).End(xlUp).Row  ' col G = PN
    lastE = ws.Cells(ws.Rows.Count, 5).End(xlUp).Row  ' col E = Id SH
    lastR = lastG
    If lastE > lastR Then lastR = lastE
    For r = 9 To lastR
        pnVal = Trim(CStr(ws.Cells(r, 7).Value))  ' col G=7 = PN
        If pnVal <> "" Then
            clVal = Trim(CStr(ws.Cells(r, 6).Value))  ' col F=6 = Cliente
            If clVal = "" Or clVal = "(seleccione)" Or clVal = "(seleccione o escriba)" Then
                nFaltantes = nFaltantes + 1
                If nFaltantes <= 10 Then
                    sinCliente = sinCliente & "  Fila " & r & ": " & pnVal & vbCrLf
                End If
            End If
        End If
    Next r
    If nFaltantes > 0 Then
        Dim msg As String
        msg = "No se puede exportar: " & nFaltantes & " l" & ChrW(237) & "nea(s) con PN sin Cliente." & vbCrLf & vbCrLf
        msg = msg & sinCliente
        If nFaltantes > 10 Then msg = msg & "  ... y " & (nFaltantes - 10) & " m" & ChrW(225) & "s" & vbCrLf
        msg = msg & vbCrLf & "Asigna un cliente en la columna F a cada l" & ChrW(237) & "nea con PN."
        MsgBox msg, vbExclamation, "Cliente faltante"
        Exit Sub
    End If

    ' V11: cliente único en COTIZACIÓN+NP
    ' Una cotización vive bajo un solo cliente; mezclar rompe el flujo.
    If esCotizacion Then
        Dim clientesUnicos As New Collection, clTmp As String
        For r = 9 To lastR
            If Trim(CStr(ws.Cells(r, 7).Value)) <> "" Then  ' V12: PN en col 7
                clTmp = Trim(CStr(ws.Cells(r, 6).Value))     ' V12: Cliente en col 6
                If clTmp <> "" Then
                    On Error Resume Next
                    clientesUnicos.Add clTmp, clTmp
                    On Error GoTo 0
                End If
            End If
        Next r
        If clientesUnicos.Count > 1 Then
            Dim lista As String, i As Long
            For i = 1 To clientesUnicos.Count
                lista = lista & "  " & ChrW(8226) & " " & clientesUnicos(i) & vbCrLf
                If i >= 6 Then
                    lista = lista & "  ... y " & (clientesUnicos.Count - i) & " m" & ChrW(225) & "s" & vbCrLf
                    Exit For
                End If
            Next i
            MsgBox "Modo COTIZACI" & ChrW(211) & "N+NP requiere UN SOLO cliente, encontr" & ChrW(233) & " " & _
                   clientesUnicos.Count & ":" & vbCrLf & vbCrLf & lista & vbCrLf & _
                   "Si quieres cargar varios clientes, usa modo SOLO_PN.", _
                   vbCritical, "Cliente mixto"
            Exit Sub
        End If
    End If

    ' V11: aviso si SOLO_PN > 2000 filas (recomienda chunkear)
    If Not esCotizacion Then
        Dim dataRows As Long
        dataRows = 0
        For r = 9 To lastR
            If Trim(CStr(ws.Cells(r, 7).Value)) <> "" Then dataRows = dataRows + 1  ' V12: col 7
        Next r
        If dataRows > 2000 Then
            Dim respChunk As VbMsgBoxResult
            respChunk = MsgBox(dataRows & " filas SOLO_PN exceden las 2,000 recomendadas." & vbCrLf & vbCrLf & _
                "Para evitar corridas largas y facilitar el resume tras crash," & vbCrLf & _
                "parte la plantilla en chunks de 2,000 antes de exportar." & vbCrLf & vbCrLf & _
                ChrW(191) & "Continuar exportando todo en un solo CSV?", _
                vbYesNo + vbExclamation, "SOLO_PN grande")
            If respChunk <> vbYes Then Exit Sub
        End If
    End If

    ' V11: sugerencia de nombre alineado con las 4 cargas
    ' Timestamp en el nombre NO afecta runKey (que se calcula sobre el contenido del CSV).
    Dim baseName As String
    If esCotizacion Then
        baseName = quoteName & "-" & Format(Now, "yyyymmdd")
    Else
        If quoteName <> "" Then
            baseName = quoteName & "-solopn-" & Format(Now, "yyyymmdd")
        Else
            baseName = "solopn-" & Format(Now, "yyyymmdd-hhnn")
        End If
    End If
    csvName = baseName

    ' V10: sanitizar caracteres inválidos en nombre de archivo
    Dim inv As Variant, ch As Variant
    inv = Array("/", "\", ":", "*", "?", Chr(34), "<", ">", "|")
    For Each ch In inv: csvName = Replace(csvName, CStr(ch), "_"): Next

    #If Mac Then
        savePath = Application.GetSaveAsFilename(csvName & ".csv")
    #Else
        savePath = Application.GetSaveAsFilename(csvName & ".csv", "CSV (*.csv), *.csv")
    #End If
    If CStr(savePath) = "False" Or savePath = "" Then Exit Sub
    If Right(savePath, 4) <> ".csv" Then savePath = savePath & ".csv"

    ' Copiar hoja a libro temporal
    ws.Copy
    Dim tmpWb As Workbook
    Set tmpWb = ActiveWorkbook

    ' V12: 71 columnas (A-BS). Eliminar columnas extra si UsedRange > 71
    Dim tmpWs As Worksheet
    Set tmpWs = tmpWb.Sheets(1)
    If tmpWs.UsedRange.Columns.Count > 71 Then
        tmpWs.Range(tmpWs.Columns(72), tmpWs.Columns(tmpWs.UsedRange.Columns.Count)).Delete
    End If

    ' V12: ordenar filas 9..lastR por (col F=6 Cliente, col G=7 PN) en libro temporal
    ' Determinismo del runKey de resume — dos exports del MISMO dataset → mismo byte-string.
    ' Si el usuario re-ordena filas entre crashes, este sort las re-canoniza antes del CSV.
    Dim tmpLastR As Long
    Dim tmpLastG As Long, tmpLastE As Long
    tmpLastG = tmpWs.Cells(tmpWs.Rows.Count, 7).End(xlUp).Row  ' col G = PN
    tmpLastE = tmpWs.Cells(tmpWs.Rows.Count, 5).End(xlUp).Row  ' col E = Id SH
    tmpLastR = tmpLastG
    If tmpLastE > tmpLastR Then tmpLastR = tmpLastE
    If tmpLastR >= 10 Then  ' al menos 2 filas de datos
        With tmpWs.Sort
            .SortFields.Clear
            .SortFields.Add Key:=tmpWs.Range("F9:F" & tmpLastR), _
                Order:=xlAscending, DataOption:=xlSortNormal
            .SortFields.Add Key:=tmpWs.Range("G9:G" & tmpLastR), _
                Order:=xlAscending, DataOption:=xlSortNormal
            .SetRange tmpWs.Range("A9:BS" & tmpLastR)
            .Header = xlNo
            .MatchCase = False
            .Orientation = xlTopToBottom
            .SortMethod = xlPinYin
            .Apply
        End With
    End If

    ' V14: predictive cols BD..BL (55..63) → "General" en el libro temporal
    ' para que el CSV UTF-8 conserve la precisión completa. La plantilla viva
    ' sigue mostrando 4 decimales al usuario (legibilidad), pero el snapshot
    ' que exportamos lleva el Value2 sin truncar.
    Dim predCol As Long
    For predCol = 55 To 63
        tmpWs.Columns(predCol).NumberFormat = "General"
    Next predCol

    ' Guardar como CSV UTF-8 (formato 62)
    Application.DisplayAlerts = False
    tmpWb.SaveAs fileName:=savePath, FileFormat:=62
    tmpWb.Close False
    Application.DisplayAlerts = True

    Dim resumen As String
    resumen = "CSV UTF-8 exportado:" & vbCrLf & savePath & vbCrLf & vbCrLf
    resumen = resumen & "Modo: " & IIf(esCotizacion, "COTIZACI" & ChrW(211) & "N+NP", "SOLO_PN") & vbCrLf
    If esCotizacion Then resumen = resumen & "Cotizaci" & ChrW(243) & "n: " & quoteName & vbCrLf
    resumen = resumen & "Filas ordenadas por (Cliente, PN) para runKey estable."
    MsgBox resumen, vbInformation, "Listo"
End Sub
