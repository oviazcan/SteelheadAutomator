' Código de la hoja Upload (v11)
' Pegar en Alt+F11 → "Hoja(Upload)" (no en un módulo)
' Dispara SombrearModoSoloPN automáticamente al cambiar H1 (la celda Modo)
'
' V11 (2026-05-27): Modo se movió de G1 a H1 (alineación spreadsheet layout v11)
' V10: escuchaba G1.

Private Sub Worksheet_Change(ByVal Target As Range)
    If Not Intersect(Target, Me.Range("H1")) Is Nothing Then
        SombrearModoSoloPN
    End If
End Sub
