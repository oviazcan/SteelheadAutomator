' === MACRO 1: ExportarCSV (v10) ===
' V10: Exporta 69 columnas (A-BQ). Layout name viene de G3 (no C4).

Sub ExportarCSV()
    Dim ws As Worksheet, csvName As String, savePath As String
    Set ws = ThisWorkbook.Sheets("Upload")

    ' V10: nombre del layout en G3 (antes C4)
    csvName = Trim(CStr(ws.Range("G3").Value))
    If csvName = "" Then csvName = "cotizacion"
    Dim inv As Variant, ch As Variant
    inv = Array("/", "\", ":", "*", "?", Chr(34), "<", ">", "|")
    For Each ch In inv: csvName = Replace(csvName, CStr(ch), "_"): Next

    ' V10: validar que toda línea con PN (col F) tenga Cliente (col E)
    ' Cliente: necesario incluso en SOLO_PN porque los PNs están scoped por customer.
    ' Proceso: NO se valida aquí porque vacío tiene significado (PN existente: copiar el suyo).
    '          La extensión valida por línea y reporta errores sin matar el batch.
    Dim r As Long, lastR As Long, sinCliente As String, nFaltantes As Long
    Dim pnVal As String, clVal As String
    lastR = ws.Cells(ws.Rows.Count, 6).End(xlUp).Row
    For r = 9 To lastR
        pnVal = Trim(CStr(ws.Cells(r, 6).Value)) ' col F = PN
        If pnVal <> "" Then
            clVal = Trim(CStr(ws.Cells(r, 5).Value)) ' col E = Cliente
            If clVal = "" Or clVal = "(seleccione)" Then
                nFaltantes = nFaltantes + 1
                If nFaltantes <= 10 Then
                    sinCliente = sinCliente & "  Fila " & r & ": " & pnVal & vbCrLf
                End If
            End If
        End If
    Next r
    If nFaltantes > 0 Then
        Dim msg As String
        msg = "No se puede exportar: " & nFaltantes & " línea(s) con PN sin Cliente." & vbCrLf & vbCrLf
        msg = msg & sinCliente
        If nFaltantes > 10 Then msg = msg & "  ... y " & (nFaltantes - 10) & " más" & vbCrLf
        msg = msg & vbCrLf & "Asigna un cliente en la columna E a cada línea con PN."
        MsgBox msg, vbExclamation, "Cliente faltante"
        Exit Sub
    End If

    #If Mac Then
        savePath = Application.GetSaveAsFilename(csvName & ".csv")
    #Else
        savePath = Application.GetSaveAsFilename(csvName & ".csv", "CSV (*.csv), *.csv")
    #End If
    If CStr(savePath) = "False" Or savePath = "" Then Exit Sub
    If Right(savePath, 4) <> ".csv" Then savePath = savePath & ".csv"

    ' Copiar hoja a libro temporal
    ws.Copy
    Dim tmpWb As Workbook
    Set tmpWb = ActiveWorkbook

    ' V10: 69 columnas (A-BQ). Eliminar columnas extra
    Dim tmpWs As Worksheet
    Set tmpWs = tmpWb.Sheets(1)
    If tmpWs.UsedRange.Columns.Count > 69 Then
        tmpWs.Range(tmpWs.Columns(70), tmpWs.Columns(tmpWs.UsedRange.Columns.Count)).Delete
    End If

    ' Guardar como CSV UTF-8 (formato 62)
    Application.DisplayAlerts = False
    tmpWb.SaveAs fileName:=savePath, FileFormat:=62
    tmpWb.Close False
    Application.DisplayAlerts = True

    MsgBox "CSV UTF-8 exportado:" & vbCrLf & savePath, vbInformation, "Listo"
End Sub
